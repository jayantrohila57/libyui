## Usage

```tsx
import { Button } from "@/components/ui/button";

export default function Example() {
  return (
    <Button>
      Submit
    </Button>
  );
}
```
 
## Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"default" \| "outline" \| "secondary" \| "ghost" \| "destructive" \| "success" \| "info" \| "link"` | `"default"` | The visual style variant of the button |
| `size` | `"default" \| "xs" \| "sm" \| "lg" \| "icon" \| "icon-xs" \| "icon-sm" \| "icon-lg"` | `"default"` | The size of the button |
| `asChild` | `boolean` | `false` | When true, the button will render as its child element using Radix UI Slot |
| `className` | `string` | - | Additional CSS classes to apply to the button |

All standard HTML button attributes are also supported.
