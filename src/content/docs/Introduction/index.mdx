<Steps>
  <Step>Install LibyUI using your preferred package manager</Step>

### Installation

```bash
npm install libyui
# or
yarn add libyui
# or
pnpm add libyui
```

### Basic Usage

```tsx
import { Button, Card } from 'libyui'

export default function App() {
  return (
    <Card className="p-6">
      <Button variant="primary">Get Started</Button>
    </Card>
  )
}
```
